function getJsonFromEndpoint(url, callback) {
  url = url + '&' + 'dbName=' + dbContext;
  console.log("Looking up :", url);
  console.log("Get url caller is: " + getJsonFromEndpoint.caller);
  fetch(url).then(function (response) {
    return response.text();
  }).then(function (response) {
    if (response) {
      var jsonstring = response.replace(/\\/g, "");

      if (jsonstring.charAt(0) === '"') {
        jsonstring = jsonstring.substring(1, jsonstring.length - 1);
      } //alert(jsonstring);


      callback(jsonstring);
    } else {
      //alert('No details found!');
      ReactDOM.render( /*#__PURE__*/React.createElement("p", null, "No records found!"), document.getElementById('dash1panel'));
    }
  }).catch(function (err) {
    return console.log(err);
  });
}
